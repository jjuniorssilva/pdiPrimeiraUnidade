1. Trabalhando com regiões

1.1 Negativo da região

Esse programa deverá solicitar ao usuário as coordenadas de dois pontos P1
e P2 localizados dentro dos limites do tamanho da imagem e exibir que lhe for fornecida. Entretanto, a região definida pelo retângulo de vértices opostos definidos pelos pontos P1 e P2 será exibida com o negativo da imagem na região correspondente:

Incluindo codigo do exemplo Hello.

negativo.py
import cv2

def get_user_points(image):
    print("Selecione dois pontos dentro dos limites da imagem.")
    while True:
        try:
            # Solicita as coordenadas de P1 e P2 ao usuário
            x1, y1 = map(int, input("Digite as coordenadas de P1 (formato: x y): ").split())
            x2, y2 = map(int, input("Digite as coordenadas de P2 (formato: x y): ").split())

            # Verifica se os pontos estão dentro dos limites da imagem
            if 0 <= x1 < image.shape[1] and 0 <= y1 < image.shape[0] and 0 <= x2 < image.shape[1] and 0 <= y2 < image.shape[0]:
                return (x1, y1), (x2, y2)
            else:
                print("Pontos fora dos limites da imagem. Tente novamente.")
        except ValueError:
            print("Entrada inválida. Por favor, digite números inteiros.")

# Carrega a imagem
image = cv2.imread("biel.png")

# Verifica se a imagem foi carregada corretamente
if image is None:
    print("Não foi possível abrir bolhas.png")
else:
    # Obtém as coordenadas dos pontos P1 e P2 do usuário
    p1, p2 = get_user_points(image)

    # Calcula as coordenadas do retângulo
    x1, y1 = min(p1[0], p2[0]), min(p1[1], p2[1])
    x2, y2 = max(p1[0], p2[0]), max(p1[1], p2[1])
    # Inverte a região do retângulo na imagem
    inverted_region = 255 - image[x1:y1, x2:y2]
    # Substitui a região invertida na imagem original
    image[x1:y1, x2:y2] = inverted_region

    # Exibe a imagem modificada
    cv2.imshow("Imagem com região invertida", image)
    cv2.waitKey(0)
    cv2.destroyAllWindows()

O código inicia com uma função chamada get_user_points, responsável pela coleta e validação dos pontos do retângulo conforme solicitado. Após a coleta, calculam-se as coordenadas do retângulo e inverte-se a região dentro desse retângulo na imagem, transformando as cores dos pixels dentro dessa área em seus complementos (255 - valor original do pixel). Em outras palavras, o código inverte as cores da região selecionada.

Por fim, com a região invertida, ela é exibida em uma janela onde o usuário pode visualizar o efeito da inversão da região retangular na imagem original

1.2 Transladando quadrantes

Neste projeto simples em Python, apresentarei o programa regions.py. Este programa interativo permite aos usuários explorar transformações básicas em imagens. Ao inserir as coordenadas de dois pontos, P1 e P2, dentro dos limites da imagem fornecida, o programa destacará em negativo a área definida por esse retângulo.

Incluindo codigo do exemplo Hello.

quadrante.py
import cv2

# Carrega a imagem
image = cv2.imread("biel.png")

# Verifica se a imagem foi carregada corretamente
if image is None:
    print("Não foi possível abrir a imagem.")
else:
    # Obtém as dimensões da imagem
    height, width, channels = image.shape

    # Calcula o ponto central da imagem
    center = (width // 2, height // 2)

    # Troca os quadrantes da diagonal principal
    temp = image[0:center[1], 0:center[0]].copy()
    image[0:center[1], 0:center[0]] = image[center[1]:, center[0]:].copy()
    image[center[1]:, center[0]:] = temp
     # Troca os quadrantes da diagonal secundaria
    temp = image[0:center[1], center[0]:].copy()
    image[0:center[1], center[0]:] = image[center[1]:, 0:center[0]].copy()
    image[center[1]:, 0:center[0]] = temp

    # Exibe a imagem com quadrantes trocados
    cv2.imshow("Imagem com Quadrantes Trocados", image)
    cv2.waitKey(0)
    cv2.destroyAllWindows()

O processo principal do código envolve a troca dos quadrantes da imagem. Ele inverte os quadrantes ao longo da diagonal principal e da diagonal secundária, dividindo a imagem em quatro partes e trocando as posições dessas partes.

Para isso, obtém-se o valor central da imagem tanto em y quanto em x. Com esse valor, é possível selecionar intervalos na matriz imagem que representam os quadrantes, permitindo destacá-los e reajustar suas posições.

2.Serialização de dados em ponto flutuante via FileStorage

O programa deverá gerar uma imagem de dimensões 256x256 pixels contendo uma senóide de 4 períodos com amplitude de 127 desenhada na horizontal, como aquela apresentada na Figura 6 . Grave a imagem no formato PNG e no formato YML. Em seguida irei comparar os arquivos gerados, extraindo uma linha de cada imagem gravada e comparando a diferença entre elas.

fileStorange.py
import cv2
import numpy as np
import matplotlib.pyplot as plt

SIDE = 256
PERIODOS = 4

def create_sine_wave_image(side_length, periods, amplitude):
    image = np.zeros((side_length, side_length), dtype=np.float32)

    for i in range(side_length):
        for j in range(side_length):
            image[i, j] = amplitude * np.sin(2 * np.pi * periods * i / side_length) + amplitude

    return image

# Cria uma senoide com 4 períodos e amplitude de 127 na horizontal
image = create_sine_wave_image(SIDE, PERIODOS, 127)

# Salva a imagem em formato PNG
png_file = "senoide_4_periodos.png"
cv2.imwrite(png_file, image)

# Salva a imagem em formato YML
yml_file = "senoide_4_periodos.yml"
fs = cv2.FileStorage(yml_file, cv2.FILE_STORAGE_WRITE)
fs.write("mat", image)
fs.release()

# Lê a imagem do arquivo YML
fs = cv2.FileStorage(yml_file, cv2.FILE_STORAGE_READ)
yml_image = fs.getNode("mat").mat()
fs.release()

# Lê a imagem no arquivo PNG
png_image = cv2.imread(png_file, cv2.IMREAD_GRAYSCALE)

# Calcula a diferença de intensidade dos pixels ao longo de uma linha horizontal no centro da imgem
row_idx = SIDE // 2 / #obtém o nnúmero que representa o meio da imagem
png_row = png_image[:,row_idx]
yml_row = yml_image[:,row_idx]

difference = np.abs(png_row - yml_row)

# Plota o gráfico da diferença calculada
plt.figure(figsize=(10, 5))
plt.plot(difference, color='b', label='Diferença')
plt.xlabel('Posição na Linha')
plt.ylabel('Diferença de Intensidade')
plt.title('Diferença de Intensidade entre PNG e YML ao longo da linha')
plt.legend()
plt.grid(True)
plt.show()

Começamos pela criação da imagem representando uma senoide com 4 períodos e amplitude de 127 na horizontal, por meio da função create_sine_wave_image. Essa função aplica a fórmula da senoide e preenche um array com os dados correspondentes. Após a geração da imagem, são realizados dois salvamentos em diferentes formatos: PNG e YML, com o intuito de comparar os dados. Isso se deve ao fato de que o formato .png possui compressão de dados, o que teoricamente pode resultar em perda de dados.

Após ler novamente as imagens, o código calcula a diferença de intensidade dos pixels entre as versões PNG e YML. Uma linha horizontal no meio da imagem é selecionada, e os valores de intensidade dos pixels ao longo dessa linha são comparados entre as versões PNG e YML. A diferença de intensidade é calculada como o valor absoluto da diferença entre esses valores de intensidade. Em seguida, é gerado um gráfico usando a biblioteca Matplotlib para visualizar a discrepância de intensidade ao longo da linha horizontal.

3.Esteganografia em imagens digitais

Escreva um programa que recupere a imagem codificada de uma imagem resultante de esteganografia. Lembre-se que os bits menos significativos dos pixels da imagem fornecida deverão compor os bits mais significativos dos pixels da imagem recuperada.

esteganografia.py
import numpy as np

def apply_reverse_steganography(image, nbits):
    height, width, _ = image.shape

    # Inicializa as imagens portadora e escondida
    carrier_image = np.zeros((height, width, 3), dtype=np.uint8)
    hidden_image = np.zeros((height, width, 3), dtype=np.uint8)

    for i in range(height):
        for j in range(width):
            val_stego = image[i, j]

            # Obtém os 3 bits menos significativos
            val_lsb = val_stego & (2**nbits - 1)

            # Obtém o restante da informação
            val_remaining = val_stego >> nbits << nbits

            # Preenche as imagens portadora e escondida
            carrier_image[i, j] = val_remaining
            hidden_image[i, j] = val_lsb << (8 - nbits)

    return carrier_image, hidden_image

nbits = 3
# Carrega a imagem esteganografada
image = cv2.imread("desafio-esteganografia.png")
if image is None:
    print("Não foi possível carregar a imagem esteganografada.")
else:
    # Aplica a reversão da esteganografia
    carrier_image, hidden_image = apply_reverse_steganography(image, nbits)

    # Salva as imagens portadora e escondida
    cv2.imwrite("imagem_portadora_recuperada.jpg", carrier_image)
    cv2.imwrite("imagem_escondida_recuperada.jpg", hidden_image)

Nesse código a princiapal parte é a função apply_reverse_steganography, que recebe uma imagem esteganografada e um número de bits nbits. Para cada pixel na imagem, ela extrai nbits bits menos significativos, preservando o restante da informação.

Para esse exemplo a quantidade de bits ocutos foram 3, assim esse bits foram extraídos e usados para criar duas imagens: uma chamada de "portadora", que retém parte da informação original, e outra chamada de "escondida", contendo os bits ocultos. A imagem portadora é criada zerando os 3 bits menos significativos, enquanto a imagem escondida é criada ajustando os 3 bits ocultos para a parte mais significativa.

E por fim o código carrega uma imagem esteganografada usando OpenCV, aplica a técnica de esteganografia reversa e salva duas imagens resultantes: uma contendo dados ocultos (imagem_escondida_recuperada.jpg) e outra preservando a informação original (imagem_portadora_recuperada.jpg).

5.1 equalizador

Implemente um programa equalize. Este deverá, para cada imagem capturada, realizar a equalização do histogram antes de exibir a imagem. Teste sua implementação apontando a câmera para ambientes com iluminações variadas e observando o efeito gerado. Assuma que as imagens processadas serão em tons de cinza.

equalizador.py

import cv2
import numpy as np


# Função para criar uma imagem do histograma com um número especificado de barras
def cria_imagem_histograma(hist, bins):
    hist_height = 256
    hist_image = np.zeros((hist_height, bins), dtype=np.uint8)

    max_val = np.max(hist)
    for b in range(256):
        bin_val = hist[b]
        height = int(bin_val * hist_height / max_val)
        cv2.line(hist_image, (b, hist_height - height), (b, hist_height), 255, 1)

    return hist_image

# Inicializa a captura de vídeo a partir da câmera
cap = cv2.VideoCapture(0)

# Verifica se a câmera está disponível
if not cap.isOpened():
    print("Câmera indisponível")
    exit()

while True:
    # Captura um frame da câmera
    ret, imagem = cap.read()

    # Converte o frame para escala de cinza
    imagem_gray = cv2.cvtColor(imagem, cv2.COLOR_BGR2GRAY)

    # Equaliza o histograma da imagem em escala de cinza
    equalizado = cv2.equalizeHist(imagem_gray)

    # Exibe o frame original
    cv2.imshow("Original", imagem)

    # Exibe o histograma do frame original
    cv2.imshow("Histograma do original", cria_imagem_histograma(cv2.calcHist([imagem_gray], [0], None, [256], [0, 256]), 256))

    # Exibe o frame equalizado
    cv2.imshow("Imagem Equalizada", equalizado)

    # Exibe o histograma do frame equalizado
    cv2.imshow("Histograma da equalizada", cria_imagem_histograma(cv2.calcHist([equalizado], [0], None, [256], [0, 256]), 256))

    # Aguarda a tecla 'Esc' para sair do loop
    if cv2.waitKey(30) >= 0:
        break

# Libera a captura de vídeo e fecha todas as janelas
cap.release()
cv2.destroyAllWindows()

Neste código, a captura de vídeo em tempo real é realizada pela biblioteca OpenCV. Primeiro, cada frame capturado é convertido para escala de cinza, simplificando a análise para uma única dimensão de intensidade.

Como sabemos, o histograma representa a distribuição dessas intensidades de pixel no frame em escala de cinza, assim o cálculo do histograma se baseia em dividir o intervalo de intensidades (0 a 255 em escala de cinza) em um número específico de "barras" ou "bins". Para cada pixel no frame, seu valor de intensidade é associado ao bin correspondente.

Ao equalizar o histograma ocorre uma redistribuição das intensidades de pixel para cobrir todo o intervalo (0 a 255), melhorando o contraste da imagem. Isso é feito para que as áreas escuras e claras na imagem sejam mais distintas e visíveis.

cap = cv2.VideoCapture(0)

    print("Câmera indisponível")

        break

    gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)

    # Calcula o histograma do frame atual
    hist = cv2.calcHist([gray], [0], None, [256], [0, 256])

    # Compara o histograma atual com o anterior (se existir), pelo metodo Qui-Quadrado Alternativo
    if prev_hist is not None:
        diff = cv2.compareHist(prev_hist, hist, cv2.HISTCMP_CHISQR)

        # Se a diferença for maior que o limiar, ativa o alarme
        if diff > limiar:
            print("Movimento detectado!")
            # Aqui você pode adicionar código para ativar um alarme ou executar outras ações

    # Atualiza o histograma anterior com o histograma atual
    prev_hist = hist

    # Exibe o frame

    # Pressione 'Esc' para sair do loop
    key = cv2.waitKey(30)
    if key == 27:
        break

# Libera a captura e fecha todas as janelas
cap.release()
cv2.destroyAllWindows()

O funcionamento desse código se inicia com a captura contínua de frames da câmera, cada um sendo convertido para tons de cinza para simplificar a análise. Em seguida, é calculado o histograma de intensidades para cada frame.

Para a detecção de movimento é comparado o histograma atual com o histograma anterior usando o método Qui-Quadrado Alternativo (Não funcionou muito bem durante a noite, sendo necessário aumentar limiar ou trocar a comparação para o BHATTACHARYYA). Se a diferença entre esses histogramas ultrapassar um limiar predefinido, nesse caso 5000, o código interpreta isso como movimento e emite a mensagem "Movimento detectado!" no console.

8. Filtragem no Domínio Espacial II

8.1 Filtro tilt_shift

A tarefa envolve a implementação de três controles deslizantes, permitindo que o usuário ajuste os parâmetros do efeito: altura da região central que entrará em foco, a taxa de decaimento e a posição vertical do centro do foco.

tilt_shift.py
import cv2
import numpy as np

# Carregar a imagem
image = cv2.imread("biel.png")

# Definir valores iniciais dos ajustes
altura_central = 150
decaimento = 0.005
posicao_vertical_centro = 300

# Função chamada quando as barras de controle são alteradas
def on_trackbar(pos):
    global altura_central, decaimento, posicao_vertical_centro
    altura_central = cv2.getTrackbarPos("Altura Central", "Tilt-Shift")
    decaimento = cv2.getTrackbarPos("Decaimento", "Tilt-Shift") / 10000.0
    posicao_vertical_centro = cv2.getTrackbarPos("Posição Vertical", "Tilt-Shift")

    # Aplica o tilt-shift
    tilt_shifted = tilt_shift(image, altura_central, decaimento, posicao_vertical_centro)

    # Mostra a imagem resultante
    cv2.imshow("Tilt-Shift", tilt_shifted)

    # Salva a imagem resultante em arquivo
    cv2.imwrite("output.jpg", tilt_shifted)

# Função de tilt-shift
def tilt_shift(image, altura_central, decaimento, posicao_vertical_centro):
    rows, cols, _ = image.shape

    # Calcula a máscara de decaimento
    mascara_decaimento = np.ones((rows, cols))
    for i in range(rows):
        mascara_decaimento[i] = 1.0 - decaimento * abs(i - posicao_vertical_centro) / altura_central
        mascara_decaimento[i] = np.clip(mascara_decaimento[i], 0, 1)

    # Aplica a máscara na imagem
    tilt_shifted = image.copy()
    for i in range(3):
        tilt_shifted[:, :, i] = image[:, :, i] * mascara_decaimento

    return tilt_shifted

# Cria janela para exibir a imagem
cv2.namedWindow("Tilt-Shift")

# Cria as barras de controle
cv2.createTrackbar("Altura Central", "Tilt-Shift", altura_central, 500, on_trackbar)
cv2.createTrackbar("Decaimento", "Tilt-Shift", int(decaimento * 10000), 10000, on_trackbar)
cv2.createTrackbar("Posição Vertical", "Tilt-Shift", posicao_vertical_centro, 500, on_trackbar)

# Chama a função on_trackbar para exibir a imagem inicial
on_trackbar(0)

cv2.waitKey(0)
cv2.destroyAllWindows()

O processo do filtro é controlado por três parâmetros ajustáveis em tempo real: "Altura Central", "Decaimento" e "Posição Vertical". Quando os controles são modificados, a função on_trackbar é acionada, recalculando o efeito Tilt-Shift.

É aplicado uma máscara de decaimento, calculada com base na distância de cada pixel à posição vertical central especificada, multiplicando essa máscara pela intensidade dos pixels na imagem original. Isso resulta em um desfoque gradual a partir da altura central definida, com a taxa de decaimento determinando a rapidez do desfoque.

O efeito final é uma região de foco nítida, delimitada pelo decaimento suave que ocorre em direção às áreas fora da altura central.

8.2 StopMotion

O objetivo é desenvolver um programa capaz de processar vídeos, aplicar o efeito Tilt-Shift em cada quadro e salvar o resultado em um novo arquivo de vídeo. O propósito é criar uma sensação de miniaturização nas cenas e para acentuar esse efeito, vamos descartar quadros em uma taxa que considerarmos apropriada, criando uma sensação de stop motion, um recurso comum em vídeos desse estilo.

tilt_shift_video.py
import cv2
import numpy as np


# Função de tilt-shift
def tilt_shift(frame, altura_central, decaimento, posicao_vertical_centro):
    rows, cols, _ = frame.shape

    # Calcula a máscara de decaimento
    mascara_decaimento = np.ones((rows, cols))
    for i in range(rows):
        mascara_decaimento[i] = 1.0 - decaimento * abs(i - posicao_vertical_centro) / altura_central
        mascara_decaimento[i] = np.clip(mascara_decaimento[i], 0, 1)

    # Aplica a máscara na imagem
    tilt_shifted_frame = frame.copy()
    for i in range(3):
        tilt_shifted_frame[:, :, i] = frame[:, :, i] * mascara_decaimento

    return tilt_shifted_frame


# Carregar o vídeo de entrada
video_path = "teste.mp4"
cap = cv2.VideoCapture(video_path)

# Verificar se o vídeo foi aberto com sucesso
if not cap.isOpened():
    print("Erro ao abrir o vídeo.")
    exit()

# Obter as informações sobre o vídeo
frame_width = int(cap.get(3))
frame_height = int(cap.get(4))
frame_rate = int(cap.get(5))
frame_count = int(cap.get(7))

# Configurações para o efeito tilt-shift
altura_central = 100
decaimento = 0.2
posicao_vertical_centro = frame_height // 2

# Configuração para o efeito stop motion
taxa_descarte_quadros = 10

# Criar um objeto VideoWriter para salvar o vídeo de saída
output_path = "output_video.mp4"
fourcc = cv2.VideoWriter_fourcc(*'mp4v')  # Codec para vídeo MP4
out = cv2.VideoWriter(output_path, fourcc, frame_rate, (frame_width, frame_height))

while True:
    ret, frame = cap.read()

    if not ret:
        break

    # Aplicar o efeito tilt-shift
    tilt_shifted_frame = tilt_shift(frame, altura_central, decaimento, posicao_vertical_centro)

    # Descartar quadros com base na taxa especificada
    for _ in range(taxa_descarte_quadros - 1):
        _, _ = cap.read()

    # Escrever o quadro processado no arquivo de saída
    out.write(tilt_shifted_frame)

    # Mostrar o quadro em tempo real
    cv2.imshow("Tilt-Shift Video", tilt_shifted_frame)

    # Pressione 'q' para sair
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

# Liberar os recursos
cap.release()
out.release()
cv2.destroyAllWindows()

Inicialmente o vídeo de entrada é carregado, e para cada quadro, é aplicado o efeito Tilt-Shift. Isso é realizado calculando uma máscara de decaimento com base na altura central, taxa de decaimento e posição vertical do centro. Essa máscara é multiplicada pelos canais de cores do quadro original, criando o efeito de desfoque gradual.

Foi incorporado tambem o efeito stop motion, descartando quadros em uma taxa específica. O vídeo processado é então salvo em um novo arquivo usando o codec MP4V.
