import cv2

def get_user_points(image):
    print("Selecione dois pontos dentro dos limites da imagem.")
    while True:
        try:
            # Solicita as coordenadas de P1 e P2 ao usu√°rio
            x1, y1 = map(int, input("Digite as coordenadas de P1 (formato: x y): ").split())
            x2, y2 = map(int, input("Digite as coordenadas de P2 (formato: x y): ").split())

            # Verifica se os pontos est√£o dentro dos limites da imagem
            if 0 <= x1 < image.shape[1] and 0 <= y1 < image.shape[0] and 0 <= x2 < image.shape[1] and 0 <= y2 < image.shape[0]:
                return (x1, y1), (x2, y2)
            else:
                print("Pontos fora dos limites da imagem. Tente novamente.")
        except ValueError:
            print("Entrada inv√°lida. Por favor, digite n√∫meros inteiros.")

    print("N√£o foi poss√≠vel abrir bolhas.png")

    # Calcula as coordenadas do ret√¢ngulo

import cv2
import numpy as np
import matplotlib.pyplot as plt

SIDE = 256
PERIODOS = 4

def create_sine_wave_image(side_length, periods, amplitude):
    image = np.zeros((side_length, side_length), dtype=np.float32)

    for i in range(side_length):
        for j in range(side_length):
            image[i, j] = amplitude * np.sin(2 * np.pi * periods * i / side_length) + amplitude

    return image

3.Esteganografia em imagens digitais

Escreva um programa que recupere a imagem codificada de uma imagem resultante de esteganografia. Lembre-se que os bits menos significativos dos pixels da imagem fornecida dever√£o compor os bits mais significativos dos pixels da imagem recuperada.

esteganografia.py
import numpy as np

def apply_reverse_steganography(image, nbits):
    height, width, _ = image.shape

    # Inicializa as imagens portadora e escondida
    carrier_image = np.zeros((height, width, 3), dtype=np.uint8)
    hidden_image = np.zeros((height, width, 3), dtype=np.uint8)

    for i in range(height):
        for j in range(width):
            val_stego = image[i, j]

            # Obt√©m os 3 bits menos significativos
            val_lsb = val_stego & (2**nbits - 1)

            # Obt√©m o restante da informa√ß√£o
            val_remaining = val_stego >> nbits << nbits

            # Preenche as imagens portadora e escondida
            carrier_image[i, j] = val_remaining
            hidden_image[i, j] = val_lsb << (8 - nbits)

    return carrier_image, hidden_image

nbits = 3
# Carrega a imagem esteganografada
image = cv2.imread("desafio-esteganografia.png")
if image is None:
    print("N√£o foi poss√≠vel carregar a imagem esteganografada.")
else:
    # Aplica a revers√£o da esteganografia
    carrier_image, hidden_image = apply_reverse_steganography(image, nbits)

    # Salva as imagens portadora e escondida
    cv2.imwrite("imagem_portadora_recuperada.jpg", carrier_image)
    cv2.imwrite("imagem_escondida_recuperada.jpg", hidden_image)

Nesse c√≥digo a princiapal parte √© a fun√ß√£o apply_reverse_steganography, que recebe uma imagem esteganografada e um n√∫mero de bits nbits. Para cada pixel na imagem, ela extrai nbits bits menos significativos, preservando o restante da informa√ß√£o.

Para esse exemplo a quantidade de bits ocutos foram 3, assim esse bits foram extra√≠dos e usados para criar duas imagens: uma chamada de "portadora", que ret√©m parte da informa√ß√£o original, e outra chamada de "escondida", contendo os bits ocultos. A imagem portadora √© criada zerando os 3 bits menos significativos, enquanto a imagem escondida √© criada ajustando os 3 bits ocultos para a parte mais significativa.

E por fim o c√≥digo carrega uma imagem esteganografada usando OpenCV, aplica a t√©cnica de esteganografia reversa e salva duas imagens resultantes: uma contendo dados ocultos (imagem_escondida_recuperada.jpg) e outra preservando a informa√ß√£o original (imagem_portadora_recuperada.jpg).

equalizador.py

import cv2
import numpy as np


# Fun√ß√£o para criar uma imagem do histograma com um n√∫mero especificado de barras
def cria_imagem_histograma(hist, bins):
    hist_height = 256
    hist_image = np.zeros((hist_height, bins), dtype=np.uint8)

    max_val = np.max(hist)
    for b in range(256):
        bin_val = hist[b]
        height = int(bin_val * hist_height / max_val)
        cv2.line(hist_image, (b, hist_height - height), (b, hist_height), 255, 1)

    return hist_image

# Inicializa a captura de v√≠deo a partir da c√¢mera
cap = cv2.VideoCapture(0)

# Verifica se a c√¢mera est√° dispon√≠vel
if not cap.isOpened():
    print("C√¢mera indispon√≠vel")
    exit()

while True:
    # Captura um frame da c√¢mera
    ret, imagem = cap.read()

    # Converte o frame para escala de cinza
    imagem_gray = cv2.cvtColor(imagem, cv2.COLOR_BGR2GRAY)

    # Equaliza o histograma da imagem em escala de cinza
    equalizado = cv2.equalizeHist(imagem_gray)

    # Exibe o frame original
    cv2.imshow("Original", imagem)

    # Exibe o histograma do frame original
    cv2.imshow("Histograma do original", cria_imagem_histograma(cv2.calcHist([imagem_gray], [0], None, [256], [0, 256]), 256))

    # Exibe o frame equalizado
    cv2.imshow("Imagem Equalizada", equalizado)

    # Exibe o histograma do frame equalizado
    cv2.imshow("Histograma da equalizada", cria_imagem_histograma(cv2.calcHist([equalizado], [0], None, [256], [0, 256]), 256))

    # Aguarda a tecla 'Esc' para sair do loop
    if cv2.waitKey(30) >= 0:
        break

# Libera a captura de v√≠deo e fecha todas as janelas
cap.release()
cv2.destroyAllWindows()

Neste c√≥digo, a captura de v√≠deo em tempo real √© realizada pela biblioteca OpenCV. Primeiro, cada frame capturado √© convertido para escala de cinza, simplificando a an√°lise para uma √∫nica dimens√£o de intensidade.

Como sabemos, o histograma representa a distribui√ß√£o dessas intensidades de pixel no frame em escala de cinza, assim o c√°lculo do histograma se baseia em dividir o intervalo de intensidades (0 a 255 em escala de cinza) em um n√∫mero espec√≠fico de "barras" ou "bins". Para cada pixel no frame, seu valor de intensidade √© associado ao bin correspondente.

Ao equalizar o histograma ocorre uma redistribui√ß√£o das intensidades de pixel para cobrir todo o intervalo (0 a 255), melhorando o contraste da imagem. Isso √© feito para que as √°reas escuras e claras na imagem sejam mais distintas e vis√≠veis.

        if diff > limiar:
            print("Movimento detectado!")
            # Aqui voc√™ pode adicionar c√≥digo para ativar um alarme ou executar outras a√ß√µes

    # Atualiza o histograma anterior com o histograma atual
    prev_hist = hist

    cv2.imshow("Frame", gray)

    # Pressione 'Esc' para sair do loop
    key = cv2.waitKey(30)
    if key == 27:
        break
